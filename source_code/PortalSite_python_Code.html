from flask import Flask, render_template, redirect, url_for

import os

app = Flask(__name__)

@app.route('/')

def index():

         return render_template('index.html')

def app_open(file):

         try:

                  os.startfile(file)

         except Exception as e:

                  print(f"エラーが発生しました: {str(e)}")

         return redirect('../')
